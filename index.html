// 오류 수정된 모바일 터치 환경 개선 코드 function setupMobileTouchEnhancements() { let longPressTimer = null; let isDragging = false; let selectedEmotion = null; // 터치 이벤트 핸들러들을 함수로 미리 정의 (이벤트 제거를 위해) const touchHandlers = new WeakMap(); return 'source'; } for (const [quadrant, emotions] of Object.entries(appState.quadrantEmotions)) { if (emotions.find(e => { // 안전한 비교를 위해 null/undefined 체크 if (!e || !emotion) return false; // 다양한 형태의 이름 비교 const eName = e.name?.ko || e.name || ''; const emotionName = emotion.name?.ko || emotion.name || ''; return eName !== emotionName; }); }); }; } if (typeof moveEmotionToSource !== 'function') { window.moveEmotionToSource = function(emotion) { removeEmotionFromAllLocations(emotion); // renderStep 함수가 존재하는지 확인 if (typeof renderStep === 'function') { renderStep(); } if (typeof updateUI === 'function') { updateUI(); } if (typeof saveToStorage === 'function') { saveToStorage(); } }; // 실제 사운드 재생 로직은 원본 코드의 sounds 객체를 참조해야 함 } }; } // 감정카드에 터치 이벤트 추가 function addTouchEvents() { const emotionChips = document.querySelectorAll('.emotion-chip'); emotionChips.forEach(chip => { // 기존 핸들러 제거 const existingHandlers = touchHandlers.get(chip); } // 새 핸들러 생성 const handlers = { start: (e) => handleTouchStart(e, chip), end: (e) => handleTouchEnd(e, chip), move: (e) => handleTouchMove(e, chip) }; // 핸들러 저장 touchHandlers.set(chip, handlers); // 이벤트 추가 chip.addEventListener('touchstart', handlers.start, { passive: false }); chip.addEventListener('touchend', handlers.end, { passive: false }); chip.addEventListener('touchmove', handlers.move, { passive: false }); }); } function handleTouchStart(e, chip) { // 터치 이벤트가 유효한지 확인 if (!e.touches || e.touches.length === 0) { console.warn('터치 이벤트에 touches 배열이 없음'); } } catch (err) { console.log('Vibration not supported'); } longPressTimer = null; }, 500); // 선택 효과 chip.classList.add('touch-pressed'); } function handleTouchMove(e, chip) { // 움직임이 감지되면 길게 누르기 취소 if (longPressTimer) { clearTimeout(longPressTimer); longPressTimer = null; } } function handleTouchEnd(e, chip) { // 타이머 정리 if (longPressTimer) { clearTimeout(longPressTimer); longPressTimer = null; // 짧은 탭 처리 if (!selectedEmotion) { selectEmotionForMove(chip); } else if (selectedEmotion === chip) { deselectEmotion(); } else { // 다른 감정카드 탭 = 위치 교환 swapEmotions(selectedEmotion, chip); } } // 선택 효과 제거 chip.classList.remove('touch-pressed'); } function selectEmotionForMove(chip) { try { // 이전 선택 해제 document.querySelectorAll('.selected-for-move').forEach(el => { el.classList.remove('selected-for-move'); }); selectedEmotion = chip; chip.classList.add('selected-for-move'); // 4분면들을 하이라이트 document.querySelectorAll('.quadrant').forEach(quadrant => { quadrant.classList.add('move-target'); // 기존 이벤트 리스너 제거 후 추가 const newHandler = (e) => handleQuadrantTap(e); quadrant.removeEventListener('click', quadrant._quadrantTapHandler); quadrant._quadrantTapHandler = newHandler; quadrant.addEventListener('click', newHandler); }); // 소스 영역도 활성화 const sourceArea = document.getElementById('emotionSource'); if (sourceArea) { sourceArea.classList.add('move-target'); const newHandler = (e) => handleSourceTap(e); sourceArea.removeEventListener('click', sourceArea._sourceTapHandler); sourceArea._sourceTapHandler = newHandler; sourceArea.addEventListener('click', newHandler); } showToast('이동할 위치를 탭하세요 (다시 탭하면 취소)'); // 자동 취소 (10초 후) setTimeout(() => { if (selectedEmotion === chip) { deselectEmotion(); } }, 10000); } catch (error) { console.error('selectEmotionForMove 오류:', error); deselectEmotion(); } } function deselectEmotion() { try { if (selectedEmotion) { selectedEmotion.classList.remove('selected-for-move'); selectedEmotion = null; } // 모든 타겟 비활성화 document.querySelectorAll('.move-target').forEach(target => { target.classList.remove('move-target'); // 저장된 핸들러 제거 if (target._quadrantTapHandler) { target.removeEventListener('click', target._quadrantTapHandler); target._quadrantTapHandler = null; } if (target._sourceTapHandler) { target.removeEventListener('click', target._sourceTapHandler); target._sourceTapHandler = null; } }); } catch (error) { console.error('deselectEmotion 오류:', error); } } function handleQuadrantTap(e) { if (!selectedEmotion) return; try { e.stopPropagation(); const quadrant = e.currentTarget.dataset.quadrant; const emotionData = selectedEmotion.dataset.emotion || selectedEmotion.closest('[data-emotion]')?.dataset.emotion; if (emotionData && quadrant) { const emotion = JSON.parse(emotionData); moveEmotionToQuadrant(emotion, quadrant); deselectEmotion(); } } catch (error) { console.error('handleQuadrantTap 오류:', error); deselectEmotion(); } } function handleSourceTap(e) { if (!selectedEmotion) return; try { e.stopPropagation(); const emotionData = selectedEmotion.dataset.emotion || selectedEmotion.closest('[data-emotion]')?.dataset.emotion; if (emotionData) { const emotion = JSON.parse(emotionData); moveEmotionToSource(emotion); deselectEmotion(); } } catch (error) { console.error('handleSourceTap 오류:', error); deselectEmotion(); } } function swapEmotions(chip1, chip2) { try { const emotion1Data = chip1.dataset.emotion || chip1.closest('[data-emotion]')?.dataset.emotion; const emotion2Data = chip2.dataset.emotion || chip2.closest('[data-emotion]')?.dataset.emotion; if (!emotion1Data || !emotion2Data) { console.warn('감정 데이터를 찾을 수 없음'); return; } const emotion1 = JSON.parse(emotion1Data); const emotion2 = JSON.parse(emotion2Data); const location1 = getCurrentEmotionLocation(emotion1); const location2 = getCurrentEmotionLocation(emotion2); // 두 감정의 위치 교환 removeEmotionFromAllLocations(emotion1); removeEmotionFromAllLocations(emotion2); if (location1 !== 'source' && typeof appState !== 'undefined' && appState.quadrantEmotions) { appState.quadrantEmotions[location1].push(emotion2); } if (location2 !== 'source' && typeof appState !== 'undefined' && appState.quadrantEmotions) { appState.quadrantEmotions[location2].push(emotion1); } showToast('감정카드 위치가 교환되었어요!'); playSound('success'); // UI 업데이트 if (typeof renderStep === 'function') renderStep(); if (typeof updateUI === 'function') updateUI(); if (typeof saveToStorage === 'function') saveToStorage(); deselectEmotion(); } catch (error) { console.error('swapEmotions 오류:', error); deselectEmotion(); } } function showMobileContextMenu(chip, touch) { try { const emotionData = chip.dataset.emotion || chip.closest('[data-emotion]')?.dataset.emotion; if (!emotionData) { console.warn('감정 데이터를 찾을 수 없음'); return; } const emotion = JSON.parse(emotionData); const menu = createMobileMenu(emotion); // 기존 메뉴 제거 document.querySelectorAll('.mobile-context-menu, .mobile-menu-overlay').forEach(m => m.remove()); // 화면 중앙에 메뉴 표시 menu.style.position = 'fixed'; menu.style.top = '50%'; menu.style.left = '50%'; menu.style.transform = 'translate(-50%, -50%)'; menu.style.zIndex = '1000'; document.body.appendChild(menu); // 배경 오버레이 추가 const overlay = document.createElement('div'); overlay.className = 'mobile-menu-overlay fixed inset-0 bg-black bg-opacity-50'; overlay.style.zIndex = '999'; overlay.addEventListener('click', () => { menu.remove(); overlay.remove(); }); document.body.appendChild(overlay); } catch (error) { console.error('showMobileContextMenu 오류:', error); } } function createMobileMenu(emotion) { const menu = document.createElement('div'); menu.className = 'mobile-context-menu fixed bg-white dark:bg-gray-800 rounded-2xl shadow-2xl p-6 max-w-sm w-full mx-4'; // 안전한 언어 접근 const language = (typeof appState !== 'undefined' && appState.language) ? appState.language : 'ko'; const currentLocation = getCurrentEmotionLocation(emotion); // 안전한 감정 이름 및 설명 접근 const emotionName = emotion.name?.[language] || emotion.name || '알 수 없는 감정'; const emotionDescription = emotion.description?.[language] || emotion.description || ''; menu.innerHTML = `
${emotion.emoji || '😐'}

${emotionName}

${emotionDescription}

${currentLocation !== 'source' ? ` ` : ''}
다른 분면으로 이동
`; // 4분면 버튼 안전하게 추가 const quadrantButtons = menu.querySelector('#quadrant-buttons'); if (typeof quadrantInfo !== 'undefined' && quadrantButtons) { Object.keys(quadrantInfo).forEach(quadrant => { if (currentLocation === quadrant) return; const info = quadrantInfo[quadrant]; const titleParts = info.title[language].split(' '); const button = document.createElement('button'); button.onclick = () => mobileMenuAction(quadrant, encodeURIComponent(JSON.stringify(emotion))); button.className = `py-3 px-2 ${info.color} border-2 border-gray-200 rounded-xl text-xs font-medium text-center`; button.innerHTML = `
${titleParts[0] || ''}
${titleParts[1] || ''}
`; quadrantButtons.appendChild(button); }); } return menu; } // 초기 설정 addTouchEvents(); // renderStep 후 재설정 (안전한 방식) if (typeof window.renderStep === 'function') { const originalRenderStep = window.renderStep; } }; } // 정리 함수 window.cleanupMobileTouchEvents = function() { // 모든 타이머 정리 if (longPressTimer) { clearTimeout(longPressTimer); longPressTimer = null; } // 선택 상태 정리 deselectEmotion(); // 모든 메뉴 제거 document.querySelectorAll('.mobile-context-menu, .mobile-menu-overlay').forEach(el => el.remove()); }; } // 모바일 메뉴 액션 함수들 (오류 처리 추가) window.mobileMenuAction = function(target, emotionJson) { try { const emotion = JSON.parse(decodeURIComponent(emotionJson)); if (target === 'source') { moveEmotionToSource(emotion); } else { moveEmotionToQuadrant(emotion, target); } closeMobileMenu(); } catch (error) { console.error('mobileMenuAction 오류:', error); closeMobileMenu(); } }; window.closeMobileMenu = function() { try { document.querySelectorAll('.mobile-context-menu, .mobile-menu-overlay').forEach(el => el.remove()); } catch (error) { console.error('closeMobileMenu 오류:', error); } }; function moveEmotionToQuadrant(emotion, quadrant) { try { // 안전성 체크 if (typeof appState === 'undefined' || !appState.quadrantEmotions) { console.error('appState.quadrantEmotions가 정의되지 않음');
